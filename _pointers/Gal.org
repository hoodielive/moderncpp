#+begin_src cpp
#include <iostream>
#include <cstdio>

using namespace std;

int main()
{

    char c = 'a';
    char* p = &c;
    char d = *p; // Derefence the pointer (return value)

    std::cout << "Assane: " << d << std::endl;

    std::cout << "The value of the deferenced pointers is: " << *p;
}

#+end_src
